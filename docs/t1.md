## wifi 万能钥匙

```python
#画wifi万能钥匙logo
import turtle as t

def border_bar(): # 边框栏
    t.up()
    t.goto(76*2, -116*2)
    t.pd()
    t.pencolor('blue')
    t.pensize(12)
    for i in range(4):
        t.circle(40*2, 90)
        t.fd(152*2)


def  signal_intensity(): #信号强度（即那四个圆弧）
    t.up()
    t.goto(-70.7,  100)
    t.pd()
    t.pencolor('blue')
    t.pensize(28)
    t.lt(225)
    t.circle(100, 90)
    t.up()
    t.circle(100, 90)
    t.pd()
    t.circle(100, 90)
    t.up()
    t.goto(-99, 130)
    t.lt(90)
    t.pd()
    t.circle(140, 90)
    t.up()
    t.circle(140, 90)
    t.pd()
    t.circle(140, 90)


def key(): # 中间部分
    t.up()
    t.goto(-5, 60)
    t.left(135)
    t.pd()
    t.pensize(40)
    t.fd(107)
    t.up()
    t.goto(32, 30)
    t.pd()
    t.fd(30)
    t.up()
    t.goto(43, -90)
    t.pd()
    t.left(90)
    t.pensize(25)
    t.pencolor('blue')
    t.lt(90)
    a=1.2
    for i in range(120):
        if 0<=i<30 or 60<=i<90:
            a=a+0.07
            t.lt(3) #向左转3度
            t.fd(a) #向前走a的步长
        else:
            a=a-0.07
            t.lt(3)
            t.fd(a)


def main():
    border_bar()
    signal_intensity()
    key()
    t.ht()
    t.done()


main()
```

## 太极图

<img align="right" src="https://i.loli.net/2020/05/16/fQIKapX49uke7MZ.png"/>

```python
import turtle as t
# 绘制“阴阳”的函数
def yinyang(radius, filling_color, inside_color):
    # 对画笔进行必要的初始化
    t.pensize(3)
    t.shape('circle')
    t.color('black', filling_color)
    # 设置填充着色标记
    t.begin_fill()
    # 绘制处在外围的“太阴”或“太阳”
    t.circle(radius/2, 180)
    t.circle(radius, 180)
    t.left(180)
    t.circle(-radius/2, 180)
    # 填充颜色结束
    t.end_fill()
    # 画笔转移
    t.left(90)
    t.penup()
    t.forward(radius*0.35)
    t.right(90)
    # 绘制处于内部的“少阴”或“少阳”
    t.pendown()
    t.color(filling_color, inside_color)
    # 填充开始
    t.begin_fill()
    t.circle(radius*0.15)
    # 填充结束
    t.end_fill()
    # 画笔转移到原点（0,0）
    t.left(90)
    t.penup()
    t.backward(radius*0.35)
    t.pendown()
    t.left(90)

# 书写文字的函数
def write_str(x, y, str):
    t.up()
    # 移动到指定的坐标
    t.goto(x, y)
    t.pd()
    t.color('gray')
    # 书写文字
    t.write(arg=str, font=('arial', 40,'normal'))

if __name__ == '__main__':
    # 调用函数绘制“太阴”和“少阳”
    yinyang(200, 'black', 'white')
    # 调用函数绘制“太阳”和“少阴”
    yinyang(200, 'white', 'black')
    # 隐藏画笔，为书写文章作准备
    t.hideturtle()
    words = "万物负阴而抱阳"
    # 调用函数书写文字
    write_str(-180, -300, words)

t.mainloop()
```
